var fib, f0, f1, i;
begin
	f0 := 0;
	f1 := 1;

	read fib;

	/* Calculate fibonacci for given fib */
	i := 2;
	while i <= fib do
	begin
		if odd i then
			f1 := f0 + f1;

		if i = 2 * (i / 2) then
			f0 := f0 + f1;

		i := i + 1;
	end;
	
	/* Print for fib > 0 */
	if fib > 0 then
	begin
		if odd i then
			write f0;
		if i = 2 * (i / 2) then
			write f1;
	end;
	
	/* Edge case where fib = 0 */
	if fib = 0 then
		write f0;
end.
